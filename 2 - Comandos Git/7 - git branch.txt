# ESTE COMANDO SIRVE PARA VER EN QUE RAMA ESTAMOS TRABAJANDO
    git branch

# ESTE COMANDO SIRVE PARA CAMBIAR EL NOMBRE DE LA RAMA MASTER
    git branch -m master main

# PARA HACER QUE EL NOMBRE DE LA RAMA PRINCIPAL DE CADA
# PROYECTO SEA SIEMPRE MAIN O NO MASTER
    git config --global init.defaultBranch <name>
    git config --global init.defaultBranch main

# CREAR UNA NUEVA RAMA
    git branch NOMBRE

# MOVERNOS A LA RAMA QUE QUEREMOS
    git checkout <nombre rama>

# UNIR RAMA AL MASTER
# HAY QUE MOVERNOS PRIMERO A LA RAMA MASTER
    git merger <nombre rama>

# BORRAR RAMA
    git branch -d <nombre rama>

# CREAR UNA RAMA Y MOVERNOS A ESE RAMA EN UN SOLO PASO
    git checkout -b <nombre rama>

# REMOVER UNA RAMA DE ALGUNA PERSONA QUE ESTA TRABAJANDO EN EL PROYECTO
# ES IMPORTANTE IR HACIENDO ESTE TIPO DE LIMPIEZAS
    git push origin <:nombre_rama>

# REVISAR RAMAS QUE YA NO EXISTE EN REMOTO PERO QUE SIGUEN EXISTIENDO EN NUESTRO LOCAL
# LAS QUE YA NO EXISTEN EN REMOTO SE VAN A ELIMINAR
    git remote prune origin
