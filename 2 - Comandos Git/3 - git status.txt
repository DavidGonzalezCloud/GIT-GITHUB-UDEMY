# PARA VER EL STATUS DE NUESTRO REPOSITORIO USAMOS:
    git status

# PARA VER EL STATUS RESUMIDO
    git status --short

# PARA AGREGAR ARCHIVOS AL TRAKER DEL REPOSITORIO USAMOS
    git add nombre_archivo

# SI QUEREMOS AGREGAR TODOS LOS ARCHIVOS USAMOS AL TRAKER
    git add .

# AGREGAMOS TODOS LOS ARCHIVOS CON EXTENSION html AL STAGE
    git add *.html

# SI QUEREMOS QUITAR ALGUN ARCHIVO DEL REPOSITORIO USAMOS
    git reset nombre_archivo
