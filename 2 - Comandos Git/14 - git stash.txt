# CREAR UN NUEVO STASH
    git stash

# CREAMOS UN NUEVO STASH CON UNA DESCRIPCION
    git stash save "descripcion del stash"

# Eliminar el stash 0
    git stash drop

# Eliminar un stash especifico
    git stash drop stash@{1}

# ELIMINAR TODOS LOS STASH
    git stash clear

# VER LA LISTA DE LOS STASH
    git stash list

# VER LISTA DE LOS STASH CON MAS INFORMACION
    git stash list  

# REGRESAMOS LA INFORMACION DEL PRIMER STASH A LA RAMA 
# TAMBIEN ELIMINA EL STASH DE LA LISTA
    git stash pop

# REGRESAMOS LA INFORMACION DE STASH ESPECIFICO A LA RAMA 
# TAMBIEN ELIMINA EL STASH DE LA LISTA
    git stash pop stash@{1}

# REGRESAMOS LA INFORMACION DEL PRIMER STASH A LA RAMA
# NO ELIMINA EL STASH DE LA LISTA
    git stash applY

# REGRESAMOS LA INFORMACION DE UN STASH ESPECIFICO A LA RAMA
# NO ELIMINA EL STASH DE LA LISTA
    git stash apply stash@{1}

# VER QUE SE HA GUARDADO EN UN STASH ESPECIFICO
    git stash show stash@{1}
