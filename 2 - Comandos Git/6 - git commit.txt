# CON ESTE COMANDO CREAMOS UN SNAPSHOOT DE LOS ARCHIVOS QUE ESTAN EN EL STAGE
     git commit -m "Comentario"


# SI QUEREMOS MODIFICAR EL COMENTARIO DEL ULTIMO COMMIT USAMOS
    git commit --amend -m "Nuevo comentario"

# SI QUEREMOS BORRAR EL ULTIMO COMMIT PARA AGREAGAR ALGO MAS HACEMOS LO SIGUIENTE
    git reset --soft HEAD^

# SI QUEREMOS BORRAR UN COMMIT ANTERIOR LO QUE TENEMOS QUE HACER ES AGREGAR EL NUMERO DESPUES DE LA FLECHITA
    git reset --soft HEAD^3

# SI QUEREMOS UNIR LOS CAMBIOS ACTUALES AL ULTIMO COMMIT USANDO
    git commit --amend --no-edit
